# vars v2 CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite all CLI commands to use the v2 `@vars/core` and `@vars/node` APIs — the new parser, resolver, codegen, crypto, and show/hide modules.

**Architecture:** The CLI (`packages/cli`) uses citty for command routing and @clack/prompts for interactive UX. Commands import from `@vars/node` (which internally uses `@vars/core`). The CLI is the only published package (`vars` on npm).

**Tech Stack:** citty 0.2.1, @clack/prompts, picocolors, @vars/node (workspace), Vitest

**Spec:** `docs/superpowers/specs/2026-03-22-vars-v2-design.md` — Section 6 (CLI Commands)

## Task 2: Context Utility Rewrite

**Files:**
- Rewrite: `packages/cli/src/utils/context.ts`

The context utility is the glue between CLI commands and the core/node modules. It handles:
- Finding `.vars` files (walk up directory tree or explicit path)
- Loading the encryption key (PIN prompt or VARS_KEY env var)
- Resolving the target environment

- [ ] **Step 1: Rewrite context.ts**

```typescript
import { resolve, dirname, join } from "node:path";
import { existsSync, readdirSync } from "node:fs";
import { getKeyFromEnv, createMasterKey, encryptMasterKey, decryptMasterKey } from "@vars/node";
import * as prompts from "@clack/prompts";

export interface CliContext {
  varsFilePath: string;
  varsDir: string;        // directory containing the .vars file
  keyFilePath: string;    // .vars/key
  env: string;
  projectRoot: string;    // git root or cwd
}

/** Find the nearest .vars file, walking up from startDir */
export function findVarsFile(startDir: string, fileName?: string): string | null {
  // If fileName provided, resolve it directly
  if (fileName) {
    const abs = resolve(startDir, fileName);
    return existsSync(abs) ? abs : null;
  }
  // Walk up looking for *.vars files
  let dir = resolve(startDir);
  while (true) {
    const files = readdirSync(dir).filter(f => f.endsWith(".vars"));
    if (files.length > 0) return resolve(dir, files[0]);
    const parent = dirname(dir);
    if (parent === dir) return null;
    dir = parent;
  }
}

/** Find all .vars files in the repo */
export function findAllVarsFiles(projectRoot: string): string[] {
  // Recursively find *.vars files, excluding node_modules, .git, dist
  const results: string[] = [];
  function walk(dir: string) {
    const entries = readdirSync(dir, { withFileTypes: true });
    for (const entry of entries) {
      if (entry.name === "node_modules" || entry.name === ".git" || entry.name === "dist") continue;
      const fullPath = join(dir, entry.name);
      if (entry.isDirectory()) walk(fullPath);
      else if (entry.name.endsWith(".vars")) results.push(fullPath);
    }
  }
  walk(projectRoot);
  return results;
}

/** Find .vars/key file, walking up from startDir */
export function findKeyFile(startDir: string): string | null {
  let dir = resolve(startDir);
  while (true) {
    const keyPath = join(dir, ".vars", "key");
    if (existsSync(keyPath)) return keyPath;
    const parent = dirname(dir);
    if (parent === dir) return null;
    dir = parent;
  }
}

/** Get encryption key — from env var or by prompting for PIN */
export async function requireKey(keyFilePath: string | null): Promise<Buffer> {
  const envKey = getKeyFromEnv();
  if (envKey) return envKey;
  if (!keyFilePath || !existsSync(keyFilePath)) {
    throw new Error("No encryption key found. Run `vars key init` first.");
  }
  const { readFileSync } = await import("node:fs");
  const encoded = readFileSync(keyFilePath, "utf8").trim();
  const pin = await prompts.password({ message: "Enter PIN:" });
  if (prompts.isCancel(pin)) process.exit(0);
  return decryptMasterKey(encoded, pin as string);
}

/** Resolve environment name with common aliases */
export function resolveEnv(env: string): string {
  const aliases: Record<string, string> = {
    development: "dev",
    production: "prod",
    staging: "staging",
  };
  return aliases[env] ?? env;
}

/** Get the git root directory */
export function getProjectRoot(startDir: string): string {
  const { execSync } = require("node:child_process");
  try {
    return execSync("git rev-parse --show-toplevel", { cwd: startDir, encoding: "utf8" }).trim();
  } catch {
    return startDir;
  }
}
```

  const outPath = filePath.replace(/\.vars$/, ".generated.ts");
  writeFileSync(outPath, code);
  console.log(pc.green(`  Generated ${outPath}`));
}

## Task 4: Run Command

**Files:**
- Rewrite: `packages/cli/src/commands/run.ts`

The run command decrypts vars in-memory and spawns a child process with env vars injected.

- [ ] **Step 1: Implement run.ts**

```typescript
import { defineCommand } from "citty";
import { resolve } from "node:path";
import { spawn } from "node:child_process";
import { resolveUseChain, getKeyFromEnv, decrypt } from "@vars/node";
import { isEncrypted } from "@vars/core";
import { findVarsFile, findKeyFile, requireKey, resolveEnv } from "../utils/context.js";
import pc from "picocolors";

export default defineCommand({
  meta: { name: "run", description: "Run a command with decrypted env vars" },
  args: {
    env: { type: "string", required: true, alias: "e", description: "Target environment" },
    param: { type: "string", description: "Param values (key=value)" },
    file: { type: "string", alias: "f", description: ".vars file" },
    _: { type: "positional", required: true, description: "Command to run" },
  },
  async run({ args, rawArgs }) {
    const env = resolveEnv(args.env);
    const params: Record<string, string> = {};
    if (args.param) {
      for (const p of (Array.isArray(args.param) ? args.param : [args.param])) {
        const [k, v] = p.split("=");
        if (k && v) params[k] = v;
      }
    }

    const file = args.file ? resolve(args.file) : findVarsFile(process.cwd());
    if (!file) { console.error(pc.red("No .vars file found")); process.exit(1); }

    // Get key for decryption
    const keyFile = findKeyFile(file);
    let key: Buffer | null = getKeyFromEnv();
    if (!key) key = await requireKey(keyFile);

    // Resolve all vars
    const resolved = resolveUseChain(file, { env, params });

    // Decrypt encrypted values
    const envVars: Record<string, string> = {};
    for (const v of resolved.vars) {
      if (v.value === undefined) continue;
      if (isEncrypted(v.value)) {
        envVars[v.flatName] = decrypt(v.value, key);
      } else {
        envVars[v.flatName] = v.value;
      }
    }

    // Find the -- separator to get the actual command
    const dashDash = rawArgs.indexOf("--");
    if (dashDash === -1) {
      console.error(pc.red("Usage: vars run --env <env> -- <command>"));
      process.exit(1);
    }
    const cmd = rawArgs.slice(dashDash + 1);
    if (cmd.length === 0) {
      console.error(pc.red("No command specified after --"));
      process.exit(1);
    }

    // Spawn child process
    const child = spawn(cmd[0], cmd.slice(1), {
      stdio: "inherit",
      env: { ...process.env, ...envVars, VARS_ENV: env },
    });

    child.on("exit", (code) => process.exit(code ?? 0));
  },
});
```

## Task 5: Check Command

**Files:**
- Rewrite: `packages/cli/src/commands/check.ts`

- [ ] **Step 1: Implement check.ts**

Validates schemas for all envs, runs check blocks, reports expired/deprecated vars, and detects stale codegen.

```typescript
import { defineCommand } from "citty";
import { resolve } from "node:path";
import { resolveUseChain, getKeyFromEnv, decrypt } from "@vars/node";
import { validateValue, evaluateCheck, isEncrypted } from "@vars/core";
import { findVarsFile, findKeyFile, requireKey, resolveEnv } from "../utils/context.js";
import pc from "picocolors";

export default defineCommand({
  meta: { name: "check", description: "Validate schemas and check blocks" },
  args: {
    file: { type: "positional", required: false },
    fresh: { type: "boolean", description: "Check if codegen is up to date" },
  },
  async run({ args }) {
    const file = args.file ? resolve(args.file) : findVarsFile(process.cwd());
    if (!file) { console.error(pc.red("No .vars file found")); process.exit(1); }

    let key: Buffer | null = getKeyFromEnv();
    const keyFile = findKeyFile(file);

    let errors = 0;
    let warnings = 0;

    // For each declared env, resolve and validate
    const resolvedDev = resolveUseChain(file, { env: "dev" });

    for (const env of resolvedDev.envs) {
      const resolved = resolveUseChain(file, { env });

      for (const v of resolved.vars) {
        if (v.value === undefined) continue;

        let value = v.value;
        if (isEncrypted(value)) {
          if (!key && keyFile) {
            try { key = await requireKey(keyFile); } catch { /* skip encrypted */ }
          }
          if (key) {
            value = decrypt(value, key);
          } else {
            continue; // skip encrypted values without key
          }
        }

        const result = validateValue(v.schema, value);
        if (!result.success) {
          console.error(pc.red(`  ✗ ${v.flatName} (${env}): ${result.issues?.[0]?.message}`));
          errors++;
        }
      }

      // Run check blocks
      if (resolved.checks.length > 0 && key) {
        const varMap: Record<string, string | undefined> = {};
        for (const v of resolved.vars) {
          let val = v.value;
          if (val && isEncrypted(val) && key) val = decrypt(val, key);
          varMap[v.name] = val;
          varMap[v.flatName] = val;
        }
        for (const check of resolved.checks) {
          for (const pred of check.predicates) {
            try {
              const passed = evaluateCheck(pred, varMap, env, {});
              if (!passed) {
                console.error(pc.red(`  ✗ check "${check.description}" failed for ${env}`));
                errors++;
              }
            } catch (e) {
              console.error(pc.red(`  ✗ check "${check.description}" error: ${e}`));
              errors++;
            }
          }
        }
      }
    }

    // Check metadata warnings
    for (const v of resolvedDev.vars) {
      if (v.metadata?.expires) {
        const expiry = new Date(v.metadata.expires);
        if (expiry < new Date()) {
          console.warn(pc.yellow(`  ⚠ ${v.flatName}: expired on ${v.metadata.expires}`));
          warnings++;
        }
      }
      if (v.metadata?.deprecated) {
        console.warn(pc.yellow(`  ⚠ ${v.flatName}: deprecated — ${v.metadata.deprecated}`));
        warnings++;
      }
    }

    // Summary
    if (errors === 0 && warnings === 0) {
      console.log(pc.green("  ✓ All checks passed"));
    } else {
      console.log(`  ${errors} error(s), ${warnings} warning(s)`);
      if (errors > 0) process.exit(1);
    }
  },
});
```

```typescript
import { defineCommand } from "citty";
import { createMasterKey, encryptMasterKey, decryptMasterKey, getKeyFromEnv } from "@vars/node";
import { readFileSync, writeFileSync, mkdirSync, existsSync } from "node:fs";
import { createHash } from "node:crypto";
import { join } from "node:path";
import { findKeyFile, getProjectRoot } from "../utils/context.js";
import * as prompts from "@clack/prompts";
import pc from "picocolors";

export default defineCommand({
  meta: { name: "key", description: "Manage encryption keys" },
  subCommands: {
    init: defineCommand({
      meta: { name: "init", description: "Create a new master key" },
      async run() {
        const root = getProjectRoot(process.cwd());
        const varsDir = join(root, ".vars");
        if (!existsSync(varsDir)) mkdirSync(varsDir, { recursive: true });

        const pin = await prompts.password({ message: "Set a PIN:" });
        if (prompts.isCancel(pin)) process.exit(0);
        const confirm = await prompts.password({ message: "Confirm PIN:" });
        if (prompts.isCancel(confirm)) process.exit(0);
        if (pin !== confirm) { console.error(pc.red("PINs do not match")); process.exit(1); }

        const key = await createMasterKey();
        const encrypted = await encryptMasterKey(key, pin as string);
        writeFileSync(join(varsDir, "key"), encrypted + "\n");
        console.log(pc.green("  Key created at .vars/key"));
      },
    }),
    fingerprint: defineCommand({
      meta: { name: "fingerprint", description: "Print key fingerprint" },
      async run() {
        const keyFile = findKeyFile(process.cwd());
        if (!keyFile) { console.error(pc.red("No key found")); process.exit(1); }
        const encoded = readFileSync(keyFile, "utf8").trim();
        const hash = createHash("sha256").update(encoded).digest("hex").slice(0, 16);
        console.log(`  Fingerprint: ${hash}`);
      },
    }),
    export: defineCommand({
      meta: { name: "export", description: "Print base64 master key for CI" },
      async run() {
        const keyFile = findKeyFile(process.cwd());
        if (!keyFile) { console.error(pc.red("No key found")); process.exit(1); }
        const encoded = readFileSync(keyFile, "utf8").trim();
        const pin = await prompts.password({ message: "Enter PIN:" });
        if (prompts.isCancel(pin)) process.exit(0);
        const key = await decryptMasterKey(encoded, pin as string);
        console.log(key.toString("base64"));
      },
    }),
  },
});
```
