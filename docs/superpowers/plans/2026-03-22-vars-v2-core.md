# vars v2 Core Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite `@vars/core` and create `@vars/node` with the v2 DSL parser, resolver, crypto, validator, codegen, and file resolution — the foundation that CLI/LSP build on top of.

**Architecture:** `@vars/core` is a pure TS computation engine (zero fs, zero native deps) that takes strings and returns structured data. `@vars/node` wraps it with filesystem access, `use`-chain resolution, and key management (Argon2, keytar). Both are internal packages — users only install the `vars` CLI.

**Tech Stack:** TypeScript 5.7, Zod 3.24, Vitest 3.0, tsup (build), pnpm workspaces, Turborepo

**Spec:** `docs/superpowers/specs/2026-03-22-vars-v2-design.md`

`packages/core/src/tokenizer.ts`:

```typescript
export enum TokenType {
  // Literals
  IDENTIFIER = "IDENTIFIER",
  KEYWORD = "KEYWORD",
  SCHEMA = "SCHEMA",
  STRING = "STRING",
  TRIPLE_STRING = "TRIPLE_STRING",
  NUMBER = "NUMBER",
  BOOLEAN = "BOOLEAN",
  ENCRYPTED = "ENCRYPTED",

  // Delimiters
  LBRACE = "LBRACE",
  RBRACE = "RBRACE",
  LPAREN = "LPAREN",
  RPAREN = "RPAREN",
  LBRACKET = "LBRACKET",
  RBRACKET = "RBRACKET",

  // Operators
  EQUALS = "EQUALS",
  COLON = "COLON",
  COMMA = "COMMA",
  DOT = "DOT",
  ARROW = "ARROW",

  // Comparison
  COMPARISON = "COMPARISON", // ==, !=, >, <, >=, <=

  // Structural
  COMMENT = "COMMENT",
  NEWLINE = "NEWLINE",
  EOF = "EOF",
  ERROR = "ERROR",
}

const KEYWORDS = new Set([
  "env", "param", "use", "group", "public", "check", "when", "else",
  "and", "or", "not", "defined", "matches", "one_of", "length", "starts_with",
]);

export interface Token {
  type: TokenType;
  value: string | number | boolean;
  line: number;
  col: number;
}

export function tokenize(source: string): Token[] {
  const tokens: Token[] = [];
  let pos = 0;
  let line = 1;
  let col = 1;
  let afterColon = false; // Track if we're in schema position

  function peek(offset = 0): string {
    return source[pos + offset] ?? "";
  }

  function advance(n = 1): string {
    const chunk = source.slice(pos, pos + n);
    for (const ch of chunk) {
      if (ch === "\n") {
        line++;
        col = 1;
      } else {
        col++;
      }
    }
    pos += n;
    return chunk;
  }

  function skipSpaces(): void {
    while (pos < source.length && (source[pos] === " " || source[pos] === "\t")) {
      advance();
    }
  }

  function readString(quote: string): Token {
    const startLine = line;
    const startCol = col;
    advance(); // skip opening quote
    let value = "";
    while (pos < source.length && source[pos] !== quote) {
      if (source[pos] === "\\") {
        advance();
        value += source[pos] ?? "";
        advance();
      } else {
        value += source[pos];
        advance();
      }
    }
    if (pos < source.length) advance(); // skip closing quote
    return { type: TokenType.STRING, value, line: startLine, col: startCol };
  }

  function readTripleString(): Token {
    const startLine = line;
    const startCol = col;
    advance(3); // skip opening """
    // Skip the first newline after opening """
    if (source[pos] === "\n") advance();
    let value = "";
    while (pos < source.length) {
      if (source[pos] === '"' && peek(1) === '"' && peek(2) === '"') {
        // Find indentation of closing """ (Kotlin convention)
        const indent = col - 1; // number of chars to strip from each line
        advance(3);
        // Strip exactly `indent` leading characters from each line
        const lines = value.split("\n");
        const stripped = lines.map((l) => l.length >= indent ? l.slice(indent) : l.trimStart());
        // Remove trailing empty line
        if (stripped.length > 0 && stripped[stripped.length - 1].trim() === "") {
          stripped.pop();
        }
        value = stripped.join("\n");
        return { type: TokenType.TRIPLE_STRING, value, line: startLine, col: startCol };
      }
      value += source[pos];
      advance();
    }
    return { type: TokenType.ERROR, value: "unterminated triple-quoted string", line: startLine, col: startCol };
  }

  function readNumber(): Token {
    const startLine = line;
    const startCol = col;
    let raw = "";
    if (source[pos] === "-") { raw += advance(); }
    while (pos < source.length && /[0-9]/.test(source[pos])) {
      raw += advance();
    }
    if (source[pos] === "." && /[0-9]/.test(peek(1))) {
      raw += advance(); // decimal
      while (pos < source.length && /[0-9]/.test(source[pos])) {
        raw += advance();
      }
    }
    return { type: TokenType.NUMBER, value: Number(raw), line: startLine, col: startCol };
  }

  function readIdentifierOrKeyword(): Token {
    const startLine = line;
    const startCol = col;
    let value = "";
    while (pos < source.length && /[a-zA-Z0-9_]/.test(source[pos])) {
      value += advance();
    }
    if (value === "true") return { type: TokenType.BOOLEAN, value: true, line: startLine, col: startCol };
    if (value === "false") return { type: TokenType.BOOLEAN, value: false, line: startLine, col: startCol };
    if (KEYWORDS.has(value)) return { type: TokenType.KEYWORD, value, line: startLine, col: startCol };
    return { type: TokenType.IDENTIFIER, value, line: startLine, col: startCol };
  }

  function readSchema(): Token {
    const startLine = line;
    const startCol = col;
    let value = "";
    let parenDepth = 0;
    let braceDepth = 0;
    let bracketDepth = 0;

    while (pos < source.length) {
      const ch = source[pos];

      // Track nesting
      if (ch === "(") parenDepth++;
      if (ch === ")") parenDepth--;
      if (ch === "{") braceDepth++;
      if (ch === "}") braceDepth--;
      if (ch === "[") bracketDepth++;
      if (ch === "]") bracketDepth--;

      // End of schema: = or { or newline at depth 0
      if (parenDepth === 0 && braceDepth === 0 && bracketDepth === 0) {
        if (ch === "=" || ch === "{" || ch === "\n" || ch === "#") {
          break;
        }
        // Also break on ( which starts metadata after schema
        if (ch === "(" && value.trim().endsWith(")")) {
          break;
        }
      }

      value += advance();
    }

    afterColon = false;
    return { type: TokenType.SCHEMA, value: value.trim(), line: startLine, col: startCol };
  }

  function readEncrypted(): Token {
    const startLine = line;
    const startCol = col;
    let value = "";
    while (pos < source.length && source[pos] !== "\n" && source[pos] !== " " && source[pos] !== "#") {
      value += advance();
    }
    return { type: TokenType.ENCRYPTED, value, line: startLine, col: startCol };
  }

  while (pos < source.length) {
    skipSpaces();
    if (pos >= source.length) break;

    const ch = source[pos];
    const startLine = line;
    const startCol = col;

    // Newline
    if (ch === "\n") {
      advance();
      afterColon = false;
      tokens.push({ type: TokenType.NEWLINE, value: "\n", line: startLine, col: startCol });
      continue;
    }

    // Comment
    if (ch === "#") {
      let comment = "";
      while (pos < source.length && source[pos] !== "\n") {
        comment += advance();
      }
      tokens.push({ type: TokenType.COMMENT, value: comment, line: startLine, col: startCol });
      continue;
    }

    // Triple-quoted string
    if (ch === '"' && peek(1) === '"' && peek(2) === '"') {
      tokens.push(readTripleString());
      continue;
    }

    // String
    if (ch === '"' || ch === "'") {
      tokens.push(readString(ch));
      continue;
    }

    // Encrypted value
    if (ch === "e" && source.slice(pos, pos + 4) === "enc:") {
      tokens.push(readEncrypted());
      continue;
    }

    // Arrow =>
    if (ch === "=" && peek(1) === ">") {
      advance(2);
      tokens.push({ type: TokenType.ARROW, value: "=>", line: startLine, col: startCol });
      continue;
    }

    // Single-char delimiters
    if (ch === "{") { advance(); tokens.push({ type: TokenType.LBRACE, value: "{", line: startLine, col: startCol }); continue; }
    if (ch === "}") { advance(); tokens.push({ type: TokenType.RBRACE, value: "}", line: startLine, col: startCol }); continue; }
    if (ch === "(") { advance(); tokens.push({ type: TokenType.LPAREN, value: "(", line: startLine, col: startCol }); continue; }
    if (ch === ")") { advance(); tokens.push({ type: TokenType.RPAREN, value: ")", line: startLine, col: startCol }); continue; }
    if (ch === "[") { advance(); tokens.push({ type: TokenType.LBRACKET, value: "[", line: startLine, col: startCol }); continue; }
    if (ch === "]") { advance(); tokens.push({ type: TokenType.RBRACKET, value: "]", line: startLine, col: startCol }); continue; }
    if (ch === ",") { advance(); tokens.push({ type: TokenType.COMMA, value: ",", line: startLine, col: startCol }); continue; }
    if (ch === ".") { advance(); tokens.push({ type: TokenType.DOT, value: ".", line: startLine, col: startCol }); continue; }

    // Colon — enters schema mode
    if (ch === ":") {
      advance();
      afterColon = true;
      tokens.push({ type: TokenType.COLON, value: ":", line: startLine, col: startCol });
      continue;
    }

    // Equals
    if (ch === "=" && peek(1) === "=") {
      advance(2);
      tokens.push({ type: TokenType.COMPARISON, value: "==", line: startLine, col: startCol });
      continue;
    }
    if (ch === "!") {
      if (peek(1) === "=") {
        advance(2);
        tokens.push({ type: TokenType.COMPARISON, value: "!=", line: startLine, col: startCol });
        continue;
      }
    }
    if (ch === ">") {
      if (peek(1) === "=") { advance(2); tokens.push({ type: TokenType.COMPARISON, value: ">=", line: startLine, col: startCol }); }
      else { advance(); tokens.push({ type: TokenType.COMPARISON, value: ">", line: startLine, col: startCol }); }
      continue;
    }
    if (ch === "<") {
      if (peek(1) === "=") { advance(2); tokens.push({ type: TokenType.COMPARISON, value: "<=", line: startLine, col: startCol }); }
      else { advance(); tokens.push({ type: TokenType.COMPARISON, value: "<", line: startLine, col: startCol }); }
      continue;
    }
    if (ch === "=") {
      advance();
      tokens.push({ type: TokenType.EQUALS, value: "=", line: startLine, col: startCol });
      continue;
    }

    // Schema (after colon, starts with z.)
    if (afterColon && ch === "z" && peek(1) === ".") {
      skipSpaces();
      tokens.push(readSchema());
      continue;
    }

    // Number
    if (/[0-9]/.test(ch) || (ch === "-" && /[0-9]/.test(peek(1)))) {
      tokens.push(readNumber());
      continue;
    }

    // Identifier or keyword
    if (/[a-zA-Z_]/.test(ch)) {
      tokens.push(readIdentifierOrKeyword());
      continue;
    }

    // Unknown character
    advance();
    tokens.push({ type: TokenType.ERROR, value: ch, line: startLine, col: startCol });
  }

  tokens.push({ type: TokenType.EOF, value: "", line, col });
  return tokens;
}
```

    expect(decrypted).toContain("prod-secret-value");
